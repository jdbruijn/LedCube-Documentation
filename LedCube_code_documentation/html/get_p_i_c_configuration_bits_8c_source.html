160  UNUSED(addrhi);

163  /*_ReadLatch: ;W0=TBLPAG,W1=Wn - data in W1:W0 */
164  asm(" mov W0,TBLPAG");
165  asm(" tblrdl [W1],W0");
166  asm(" tblrdh [W1],W1");
167  /* This gives a compiler warning because not explicitly returning a value
168  * from the function, but since this is assembly, do not know how to prevent
169  * warning.
170  * The function is correct, since W1:W0 being used to return the tblrdl,
171  * tblrdh. Could write less efficient code and use a temporary UWord32.

173 }

193 void
194 getPICConfigBits(void)
195 {
196  /* Base address of CW1: Flash Configuration Word 1 */
197  _GPCB_uReg32 CW1Addr;
198  /* Base address of CW1: Flash Configuration Word 2 */
199  _GPCB_uReg32 CW2Addr;
200  /* Value of CW1: Flash Configuration Word 1 */
201  CWReg1_t CW1;
202  /* Value of CW1: Flash Configuration Word 2 */
203  CWReg2_t CW2;
204 
205  /* For the Configuration Word Addresses see PIC24FJ64GA004 Family Data
206  * Sheet.pdf (page 209, TABLE 24-1).
207  */
208  CW1Addr.Val32 = _CONFIG1_BASE_ADDRESS;
209  CW2Addr.Val32 = _CONFIG2_BASE_ADDRESS;
210 
211  /********** Read and verify Flash Configuration Word 1 ********************/
212  uprintf("\n\nReading Flash Configuration Word 1...\n");
213  CW1.Val32 = _GPCB_readLatch(CW1Addr.Word.HW,
214  CW1Addr.Word.LW);
215  uprintf("Binary read value:\n");
216  uputbits(CW1.Val32, 24);
217 
218  if (_GPCB_verifyConfigUpperByte(CW1.Val32) == true) {
219  uprintf("\nSuccessfully verified the Flash Configuration Word 1" \
220  " with address 0x%.2X%.4X\n", CW1Addr.Word.HW, CW1Addr.Word.LW);
221 
222  /********** Document Flash Configuration Word 1 results ***************/
223  /* JTAGEN (bit 14) */
224  PRINT_DESC(JTAGEN_DESC);
225  if (CW1.CW1bits.JTAGEN) {
226  PRINT_RESULT(JTAGEN_1);
227  } else {
228  PRINT_RESULT(JTAGEN_0);
229  }
230  /* GCP (bit 13) */
231  PRINT_DESC(GCP_DESC);
232  if (CW1.CW1bits.GCP) {
233  PRINT_RESULT(GCP_1);
234  } else {
235  PRINT_RESULT(GCP_0);
236  }
237  /* GWRP (bit 12) */
238  PRINT_DESC(GWRP_DESC);
239  if (CW1.CW1bits.GWRP) {
240  PRINT_RESULT(GWRP_1);
241  } else {
242  PRINT_RESULT(GWRP_0);
243  }
244  /* !DEBUG (bit 11) */
245  PRINT_DESC(BKBUG_DESC);
246  if (CW1.CW1bits.BKBUG) {
247  PRINT_RESULT(BKBUG_1);
248  } else {
249  PRINT_RESULT(BKBUG_0);
250  }
251  /* ICS (bit 9-8) */
252  PRINT_DESC(ICS_DESC);
253  if (CW1.CW1bits.ICS == 0b11) {
254  PRINT_RESULT(ICS_11);
255  } else if (CW1.CW1bits.ICS == 0b10) {
256  PRINT_RESULT(ICS_10);
257  } else if (CW1.CW1bits.ICS == 0b01) {
258  PRINT_RESULT(ICS_01);
259  } else if (CW1.CW1bits.ICS == 0b00) {
260  PRINT_RESULT(ICS_00);
261  }
262  /* FWDTEN (bit 7) */
263  PRINT_DESC(FWDTEN_DESC);
264  if (CW1.CW1bits.FWDTEN) {
265  PRINT_RESULT(FWDTEN_1);
266  } else {
267  PRINT_RESULT(FWDTEN_0);
268  }
269  /* WINDIS (bit 6) */
270  PRINT_DESC(WINDIS_DESC);
271  if (CW1.CW1bits.WINDIS) {
272  PRINT_RESULT(WINDIS_1);
273  } else {
274  if (CW1.CW1bits.FWDTEN) {
275  PRINT_RESULT(WINDIS_0);
276  } else {
277  PRINT_RESULT(WINDIS_0);
278  uprintf(" DEBUG: %s:%d:%s(): WARNING: FWDTEN is not '1', " \
279  "this setting has no effect!",
280  __FILE__, __LINE__, __FUNCTION__);
281  }
282 
283  }
284  /* FWPSA (bit 4) */
285  PRINT_DESC(FWPSA_DESC);
286  if (CW1.CW1bits.FWPSA) {
287  PRINT_RESULT(FWPSA_1);
288  } else {
289  PRINT_RESULT(FWPSA_0);
290  }
291  /* WDTPS (bit 3-0) */
292  PRINT_DESC(WDTPS_DESC);
293  if (CW1.CW1bits.WDTPS == 0b1111) {
294  PRINT_RESULT(WDTPS_1111);
295  } else if (CW1.CW1bits.WDTPS == 0b1110) {
296  PRINT_RESULT(WDTPS_1110);
297  } else if (CW1.CW1bits.WDTPS == 0b1101) {
298  PRINT_RESULT(WDTPS_1101);
299  } else if (CW1.CW1bits.WDTPS == 0b1100) {
300  PRINT_RESULT(WDTPS_1100);
301  } else if (CW1.CW1bits.WDTPS == 0b1011) {
302  PRINT_RESULT(WDTPS_1011);
303  } else if (CW1.CW1bits.WDTPS == 0b1010) {
304  PRINT_RESULT(WDTPS_1010);
305  } else if (CW1.CW1bits.WDTPS == 0b1001) {
306  PRINT_RESULT(WDTPS_1001);
307  } else if (CW1.CW1bits.WDTPS == 0b1000) {
308  PRINT_RESULT(WDTPS_1000);
309  } else if (CW1.CW1bits.WDTPS == 0b0111) {
310  PRINT_RESULT(WDTPS_0111);
311  } else if (CW1.CW1bits.WDTPS == 0b0110) {
312  PRINT_RESULT(WDTPS_0110);
313  } else if (CW1.CW1bits.WDTPS == 0b0101) {
314  PRINT_RESULT(WDTPS_0101);
315  } else if (CW1.CW1bits.WDTPS == 0b0100) {
316  PRINT_RESULT(WDTPS_0100);
317  } else if (CW1.CW1bits.WDTPS == 0b0011) {
318  PRINT_RESULT(WDTPS_0011);
319  } else if (CW1.CW1bits.WDTPS == 0b0010) {
320  PRINT_RESULT(WDTPS_0010);
321  } else if (CW1.CW1bits.WDTPS == 0b0001) {
322  PRINT_RESULT(WDTPS_0001);
323  } else if (CW1.CW1bits.WDTPS == 0b0000) {
324  PRINT_RESULT(WDTPS_0000);
325  }
326  } else {
327  uprintf("\nVerifying Flash Configuration Word 1 with address" \
328  " 0x%.2X %.4X failed...\n", CW1Addr.Word.HW, CW1Addr.Word.LW);
329  }
330  uprintf("Done Reading Flash Configuration Word 1.\n");
331 
332  /********** Read and verify Flash Configuration Word 2 ********************/
333  uprintf("\nReading Flash Configuration Word 2...\n");
334  CW2.Val32 = _GPCB_readLatch(CW2Addr.Word.HW,
335  CW2Addr.Word.LW);
336  uprintf("Binary read value:\n");
337  uputbits(CW2.Val32, 24);
338 
339  if (_GPCB_verifyConfigUpperByte(CW2.Val32) == true) {
340  uprintf("\nSuccessfully verified the Flash Configuration Word 2" \
341  " with address 0x%.2X%.4X\n", CW2Addr.Word.HW, CW2Addr.Word.LW);
342 
343  /********** Document Flash Configuration Word 2 results ***************/
344  /* IESO (bit 15) */
345  PRINT_DESC(IESO_DESC);
346  if (CW2.CW2bits.IESO) {
347  PRINT_RESULT(IESO_1);
348  } else {
349  PRINT_RESULT(IESO_0);
350  }
351  /* WUTSEL (bit 14-13) */
352  PRINT_DESC(WUTSEL_DESC);
353  if (CW2.CW2bits.WUTSEL == 0b11) {
354  PRINT_RESULT(WUTSEL_11);
355  } else if (CW2.CW2bits.WUTSEL == 0b01) {
356  PRINT_RESULT(WUTSEL_01);
357  } else {
358  PRINT_RESULT(WUTSEL_x0);
359  }
360  /* SOSCSEL (bit 12-11) */
361  PRINT_DESC(SOSCSEL_DESC);
362  if (CW2.CW2bits.SOSCSEL == 0b11) {
363  PRINT_RESULT(SOSCSEL_11);
364  } else if (CW2.CW2bits.SOSCSEL == 0b01) {
365  PRINT_RESULT(SOSCSEL_01);
366  } else {
367  PRINT_RESULT(SOSCSEL_x0);
368  }
369  /* FNOSC (bit 10-8) */
370  PRINT_DESC(FNOSC_DESC);
371  if (CW2.CW2bits.FNOSC == 0b111) {
372  PRINT_RESULT(FNOSC_111);
373  } else if (CW2.CW2bits.FNOSC == 0b110) {
374  PRINT_RESULT(FNOSC_110);
375  } else if (CW2.CW2bits.FNOSC == 0b101) {
376  PRINT_RESULT(FNOSC_101);
377  } else if (CW2.CW2bits.FNOSC == 0b100) {
378  PRINT_RESULT(FNOSC_100);
379  } else if (CW2.CW2bits.FNOSC == 0b011) {
380  PRINT_RESULT(FNOSC_011);
381  } else if (CW2.CW2bits.FNOSC == 0b010) {
382  PRINT_RESULT(FNOSC_010);
383  } else if (CW2.CW2bits.FNOSC == 0b001) {
384  PRINT_RESULT(FNOSC_001);
385  } else if (CW2.CW2bits.FNOSC == 0b000) {
386  PRINT_RESULT(FNOSC_000);
387  }
388  /* FCKSM (bit 7-6) */
389  PRINT_DESC(FCKSM_DESC);
390  if (CW2.CW2bits.FCKSM == 0b01) {
391  PRINT_RESULT(FCKSM_01);
392  } else if (CW2.CW2bits.FCKSM == 0b00) {
393  PRINT_RESULT(FCKSM_00);
394  } else {
395  PRINT_RESULT(FCKSM_1x);
396  }
397  /* OSCIOFCN (bit 5) */
398  PRINT_DESC(OSCIOFCN_DESC);
399  if (CW2.CW2bits.POSCMD == 0b11 || CW2.CW2bits.POSCMD == 00) {
400  if (CW2.CW2bits.OSCIOFCN) {
401  PRINT_RESULT(OSCIOFCN_1);
402  } else {
403  PRINT_RESULT(OSCIOFCN_0);
404  }
405  } else if (CW2.CW2bits.POSCMD == 0b10 || CW2.CW2bits.POSCMD == 01) {
406  PRINT_RESULT(OSCIOFCN_x);
407  }
408  /* IOL1WAY (bit 4) */
409  PRINT_DESC(IOL1WAY_DESC);
410  if (CW2.CW2bits.IOL1WAY) {
411  PRINT_RESULT(IOL1WAY_1);
412  } else {
413  PRINT_RESULT(IOL1WAY_0);
414  }
415  /* I2C1SEL (bit 2) */
416  PRINT_DESC(I2C1SEL_DESC);
417  if (CW2.CW2bits.I2C1SEL) {
418  PRINT_RESULT(I2C1SEL_1);
419  } else {
420  PRINT_RESULT(I2C1SEL_0);
421  }
422  /* POSCMD (bit 1-0) */
423  PRINT_DESC(POSCMD_DESC);
424  if (CW2.CW2bits.POSCMD == 0b11) {
425  PRINT_RESULT(POSCMD_11);
426  } else if (CW2.CW2bits.POSCMD == 0b10) {
427  PRINT_RESULT(POSCMD_10);
428  } else if (CW2.CW2bits.POSCMD == 0b01) {
429  PRINT_RESULT(POSCMD_01);
430  } else if (CW2.CW2bits.POSCMD == 0b00) {
431  PRINT_RESULT(POSCMD_00);
432  }
433  } else {
434  uprintf("\nVerifying Flash Configuration Word 2 with address" \
435  " 0x%.2X %.4X failed...\n", CW2Addr.Word.HW, CW2Addr.Word.LW);
436  }
437  uprintf("Done Reading Flash Configuration Word 2.\n\n\n");
438 
439  return;
440 }
