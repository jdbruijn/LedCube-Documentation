39 void
40 LedCube_setPixel( uint8_t const _x, uint8_t const _y, uint8_t const _z,
41  uint8_t const _red, uint8_t const _green, uint8_t const _blue ) {
42  DEBUG_PRINTF_FUNCTION_CALL("%u, %u, %u, %u, %u, %u", \
43  _x, _y, _z, _red, _green, _blue);
44 
45  /********** Check conditions **********************************************/
46  /* X, y, z positions within range */
47  ASSERT( LEDCUBE_MIN_XYZ <= _x && \
48  LEDCUBE_MAX_XYZ >= _x );
49  ASSERT( LEDCUBE_MIN_XYZ <= _y && \
50  LEDCUBE_MAX_XYZ >= _y );
51  ASSERT( LEDCUBE_MIN_XYZ <= _z && \
52  LEDCUBE_MAX_XYZ >= _z );
53  /* Intensities within range */
54  ASSERT( LEDCUBE_MIN_INTENSITY <= _red && \
55  LEDCUBE_MAX_INTENSITY >= _red );
56  ASSERT( LEDCUBE_MIN_INTENSITY <= _green && \
57  LEDCUBE_MAX_INTENSITY >= _green );
58  ASSERT( LEDCUBE_MIN_INTENSITY <= _blue && \
59  LEDCUBE_MAX_INTENSITY >= _blue );
60 
61  /********** Calculate position ********************************************/
62  /* _x Ranges from 0 to 7 and selects the row, 0 for the back row and
63  * 7 for the front row.
64  * _y Ranges from 0 to 7 and selects the column, 0 for the left column
65  * and 7 for the right column.
66  * _z Ranges from 0 to 7 and selects the layer, 0 for the bottom layer
67  * and 7 for the top layer.
68  *
69  * @Note It's necessary that the variables where the colour values are
70  * AND-ed with have the prefix UL, since an 32-bits number is needed for
71  * further calculations.
72  *
73  * Assume the LED intensity of 15 is written to a the LED in the first
74  * column. Than the binary value of that LED's intensity will be 00001111.
75  * Since a BAM of four bits is used the zero padding is not relevant.
76  * This means that every BAM-round a value of 1 needs to be written to the
77  * LED. As said earlier, if this is written to the fist column it should
78  * look as the sample below the next paragraph.
79  * So the first (rightmost) bit of the LED intensity does not need to be
80  * shifted left since it is already in the right position. The second bit
81  * (from the right) needs to be shifted 7 positions to the left since it is
82  * in the second position and needs to be in the 9th. The third bit is needs
83  * to be shifted 14 positions to the left since it is in the third position
84  * and needs to be in the 17th. And so on...
85  * Example:
86  * BAM-round: | 3 || 2 || 1 || 0 |
87  * 00000001000000010000000100000001
88  * LED intensity 1 (same position)
89  * | 1 (7 positions to the left)
90  * | 1 (14 positions to the left)
91  * | 1 (21 positions to the left)
92  */
93  pCubeData_t const pCubeDataWrite = pCubeControlData->pCubeDataWrite;
94  (pCubeDataWrite + _z * CUBEDATA_MAX_Z_C + _x)->red |= \
95  ((_red & 0x01UL) << _y) |
96  ((_red & 0x02UL) << (_y + 7)) |
97  ((_red & 0x04UL) << (_y + 14)) |
98  ((_red & 0x08UL) << (_y + 21));
99  (pCubeDataWrite + _z * CUBEDATA_MAX_Z_C + _x)->green |= \
100  ((_green & 0x01UL) << _y) |
101  ((_green & 0x02UL) << (_y + 7)) |
102  ((_green & 0x04UL) << (_y + 14)) |
103  ((_green & 0x08UL) << (_y + 21));
104  (pCubeDataWrite + _z * CUBEDATA_MAX_Z_C + _x)->blue |= \
105  ((_blue & 0x01UL) << _y) |
106  ((_blue & 0x02UL) << (_y + 7)) |
107  ((_blue & 0x04UL) << (_y + 14)) |
108  ((_blue & 0x08UL) << (_y + 21));
109 
110  return;
111 }
