136 void
137 Uart1_putNum( int32_t _num, uint8_t _base, bool _printBase ) {
138  char const digit[] = "0123456789ABCDEF";
139  char str[20] = {0};
140 
141  char* p_str = str;
142  if( _num < 0 ) {
143  *p_str++ = '-'; // Add minus prefix
144  _num *= -1; // Make number positive for further calculations
145  }
146 
155  uint32_t shifter = _num;
156  do {
157  ++p_str;
158  shifter /= _base;
159  } while( shifter );
160 
161  if( _printBase && _base == 2 ) {
162  *p_str = 'b';
163  *++p_str = '\0';
164  p_str--;
165  }
166  else if( _printBase && _base == 16 ) {
167  *p_str = 'h';
168  *++p_str = '\0';
169  p_str--;
170  }
171  else {
172  *p_str = '\0';
173  }
174 
180  do {
181  *--p_str = digit[_num % _base];
182  _num /= _base;
183  } while( _num );
184 
185  Uart1_puts(str);
186 
187  return;
188 }

243 void
244 Uart1_gets( char *_str, uint8_t _num ) {
245  uint8_t i = 0;
246  while(i < _num) {
247  if(pRxBuffer->bufferOverflowFlag == 1) {
252 #ifdef DEBUG
253  Uart1_printf("\n\n ERROR: %s:%d:%s(): " \
254  "UART1 FIFO software buffer overflow\n\n",
255  __FILE__, __LINE__, __FUNCTION__);
256 #else /* DEBUG */
257  Uart1_puts("\n\n ERROR: "); Uart1_puts(__FILE__);
258  Uart1_putNum(__LINE__, 10, false); Uart1_puts(__FUNCTION__);
259  Uart1_puts("UART1 FIFO software buffer overflow\n\n");
260 #endif /* DEBUG */
261 
262  return;
263  } else if(pRxBuffer->bufferNotEmptyFlag) {
264  *_str = swFifoBufferGet( pRxBuffer );
265 
266  if(*_str == UART_CR || *_str == UART_LF) {
267  *_str = '\0'; // Ignore character and end the string
268  return; // String is complete, exit loop and return
269  }
270 
271  _str++; // Move to next position in the string
272  }
273  i++;
274  }
275 
276  return;
277 }
