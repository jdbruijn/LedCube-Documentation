48 void
49 Uart1_init(void)
50 {
51  /********** Configure UARTx Module ****************************************/
52  U1MODE = 0; // Reset UARTx module
53  /* "Baud Rate = [Fp / (16 x (UxBRG + 1))]
54  * UxBRG = [(Fp / (16 x Baud Rate)) - 1]" (Universal Asynchronous Receiver
55  * Transmitter (UART), Equation 3-1: UARTx Baud Rate (BRGH = 0), Microchip)
56  */
57  U1BRG = UART1_BRG;
58 
59  /* 11 = UxTX, UxRX and BCLKx pins are enabled and used; UxCTS pin is
60  * controlled by port latches
61  * 10 = UxTX, UxRX, UxCTS and UxRTS pins are enabled and used
62  * 01 = UxTX, UxRX and UxRTS pins are enabled and used; UxCTS pin is
63  * controlled by port latches
64  * 00 = UxTX and UxRX pins are enabled and used; UxCTS, UxRTS and BCLKx
65  * pins are controlled by port latches
66  */
67  U1MODEbits.UEN = 0b10;
68 
69  /* Note: The calculation of UxBRG depends on this setting!
70  * 1 = BRG generates 4 clocks per bit period (4x baud clock, High-Speed
71  * mode)
72  * 0 = BRG generates 16 clocks per bit period (16x baud clock, Standard
73  * Speed mode)
74  */
75  U1MODEbits.BRGH = 1;
76 
77  /* 11 = 9-bit data, no parity
78  * 10 = 8-bit data, odd parity
79  * 01 = 8-bit data, even parity
80  * 00 = 8-bit data, no parity
81  */
82  U1MODEbits.PDSEL = 0b00;
83  /* 1 = 2 Stop bits
84  * 0 = 1 Stop bit
85  */
86  U1MODEbits.STSEL = 0;
87 
88  /********** Configure interrupts ******************************************/
89  IPC2bits.U1RXIP = INT_U1RX_PRIO; // Set UARTx RX interrupt priority
90  IFS0bits.U1RXIF = 0; // Clear the UARTx RX interrupt flag
91  IEC0bits.U1RXIE = 1; // Enable UARTx RX interrupts
92  /* 11 = Interrupt flag bit is set when the receive buffer is full
93  * (i.e., 4 data characters)
94  * 10 = Interrupt flag bit is set when the receive buffer is 3/4 full
95  * (i.e., 3 data characters)
96  * 0x = Interrupt flag bit is set when a character is received
97  */
98  U1STAbits.URXISEL = 0b00;
99 
100  U1MODEbits.UARTEN = 1; // Enable UARTx module
101  U1STAbits.UTXEN = 1; // Enable UARTx transmitter
102  DELAY_US(105);
103 
104 #ifdef DEBUG
105  uprintf(" DEBUG: %s(): Initialize complete...\n"
106  " DEBUG: %s(): Baud Rate Error (must be below 5%%): %.2f%%\n",
107  __FUNCTION__, __FUNCTION__, (UART1_BAUD_ERROR / 100.0));
108 #endif
109 
110  return;
111 }

132 void
133 Uart1_putNum(int32_t _num, uint8_t _base, bool _printBase)
134 {
135  char const digit[] = "0123456789ABCDEF";
136  char str[20] = {0};
137 
138  char* p_str = str;
139  if (_num < 0) {
140  *p_str++ = '-'; // Add minus prefix
141  _num *= -1; // Make number positive for further calculations
142  }
143 
144  /* Calculate how much char positions are needed for the number string. This
145  * moves from left to right and puts the NUL terminator at the end of the
146  * number string. It also adds a prefix to binary and hexadecimal numbers.
147  * At the end the pointer is at the end of the number string (rightmost
148  * character) or, in case of binary and hexadecimal numbers, one to the
149  * rightmost character.
150  */
151  uint32_t shifter = _num;
152  do {
153  ++p_str;
154  shifter /= _base;
155  } while (shifter);
156 
157  if (_printBase && _base == 2) {
158  *p_str = 'b';
159  *++p_str = '\0';
160  p_str--;
161  } else if (_printBase && _base == 16) {
162  *p_str = 'h';
163  *++p_str = '\0';
164  p_str--;
165  } else {
166  *p_str = '\0';
167  }
168 
169  /* Move back along the number string (from right to left) and insert the
170  * digits on the way. In case of the number 123 the 3 will be inserted
171  * first, since this is the rightmost digit.
172  */
173  do {
174  *--p_str = digit[_num % _base];
175  _num /= _base;
176  } while (_num);
177 
178  uputs(str);
179 
180  return;
181 }

183 void
184 Uart1_putBits(const uint64_t _num, uint8_t _nBits)
185 {
186  uint8_t i;
187 
188  // Comparing with 8 instead of 7, due to overflow of unsigned.
189  for (i = _nBits; i > 0; i--) {
190  /* Output plus 48, since character '0' is in the 48th place of the
191  * ASCII table.
192  */
193  uputc(((_num >> (i - 1)) & 1) + 48);
194  }
195 
196  return;
197 }

222 void
223 Uart1_gets(char *_str, uint8_t _num)
224 {
225  uint8_t i = 0;
226  while (i < _num) {
227  if (pRxBuffer->bufferOverflowFlag == 1) {
228  /* Handle overflow error by (at this time) informing about the
229  * overflow.
230  */
231 #ifdef DEBUG
232  uprintf("\n\n ERROR: %s:%d:%s(): " \
233  "UART1 FIFO software buffer overflow\n\n",
234  __FILE__, __LINE__, __FUNCTION__);
235 #else /* DEBUG */
236  uputs("\n\n ERROR: ");
237  uputs(__FILE__);
238  uputnum(__LINE__, 10, false);
239  uputs(__FUNCTION__);
240  uputs("UART1 FIFO software buffer overflow\n\n");
241 #endif /* DEBUG */
242 
243  return;
244  } else if (pRxBuffer->bufferNotEmptyFlag) {
245  *_str = swFifoBufferGet(pRxBuffer);
246 
247  if (*_str == UART_CR || *_str == UART_LF) {
248  *_str = '\0'; // Ignore character and end the string
249  return; // String is complete, exit loop and return
250  }
251 
252  _str++; // Move to next position in the string
253  }
254  i++;
255  }
256 
257  return;
258 }
