37 void
38 Uart1_init( void ) {
39  /********** Configure UARTx Module ****************************************/
40  U1MODE = 0; // Reset UARTx module
41  /* "Baud Rate = [Fp / (16 x (UxBRG + 1))]
42  * UxBRG = [(Fp / (16 x Baud Rate)) - 1]" (Universal Asynchronous Receiver
43  * Transmitter (UART), Equation 3-1: UARTx Baud Rate (BRGH = 0), Microchip)
44  */
45  U1BRG = UART1_BRG;
46 
47  /* 11 = UxTX, UxRX and BCLKx pins are enabled and used; UxCTS pin is
48  * controlled by port latches
49  * 10 = UxTX, UxRX, UxCTS and UxRTS pins are enabled and used
50  * 01 = UxTX, UxRX and UxRTS pins are enabled and used; UxCTS pin is
51  * controlled by port latches
52  * 00 = UxTX and UxRX pins are enabled and used; UxCTS, UxRTS and BCLKx
53  * pins are controlled by port latches
54  */
55  U1MODEbits.UEN = 0b10;
56 
57  /* Note: The calculation of UxBRG depends on this setting!
58  * 1 = BRG generates 4 clocks per bit period (4x baud clock, High-Speed
59  * mode)
60  * 0 = BRG generates 16 clocks per bit period (16x baud clock, Standard
61  * Speed mode)
62  */
63  U1MODEbits.BRGH = 1;
64 
65  /* 11 = 9-bit data, no parity
66  * 10 = 8-bit data, odd parity
67  * 01 = 8-bit data, even parity
68  * 00 = 8-bit data, no parity
69  */
70  U1MODEbits.PDSEL = 0b00;
71  /* 1 = 2 Stop bits
72  * 0 = 1 Stop bit
73  */
74  U1MODEbits.STSEL = 0;
75 
76  /********** Configure interrupts ******************************************/
77  IPC2bits.U1RXIP = INT_U1RX_PRIO; // Set UARTx RX interrupt priority
78  IFS0bits.U1RXIF = 0; // Clear the UARTx RX interrupt flag
79  IEC0bits.U1RXIE = 1; // Enable UARTx RX interrupts
80  /* 11 = Interrupt flag bit is set when the receive buffer is full
81  * (i.e., 4 data characters)
82  * 10 = Interrupt flag bit is set when the receive buffer is 3/4 full
83  * (i.e., 3 data characters)
84  * 0x = Interrupt flag bit is set when a character is received
85  */
86  U1STAbits.URXISEL = 0b00;
87 
88  U1MODEbits.UARTEN = 1; // Enable UARTx module
89  U1STAbits.UTXEN = 1; // Enable UARTx transmitter
90  Delay_us(105);
91 
92 #ifdef DEBUG
93  Uart1_printf(" DEBUG: %s(): Initialize complete...\n"
94  " DEBUG: %s(): Baud Rate Error (must be below 5%%): %.2f%%\n",
95  __FUNCTION__, __FUNCTION__, (UART1_BAUD_ERROR / 100.0) );
96 #endif
97 
98  return;
99 }

118 void
119 Uart1_putNum( int32_t _num, uint8_t _base, bool _printBase ) {
120  char const digit[] = "0123456789ABCDEF";
121  char str[20] = {0};
122 
123  char* p_str = str;
124  if( _num < 0 ) {
125  *p_str++ = '-'; // Add minus prefix
126  _num *= -1; // Make number positive for further calculations
127  }
128 
129  /* Calculate how much char positions are needed for the number string. This
130  * moves from left to right and puts the NUL terminator at the end of the
131  * number string. It also adds a prefix to binary and hexadecimal numbers.
132  * At the end the pointer is at the end of the number string (rightmost
133  * character) or, in case of binary and hexadecimal numbers, one to the
134  * rightmost character.
135  */
136  uint32_t shifter = _num;
137  do {
138  ++p_str;
139  shifter /= _base;
140  } while( shifter );
141 
142  if( _printBase && _base == 2 ) {
143  *p_str = 'b';
144  *++p_str = '\0';
145  p_str--;
146  }
147  else if( _printBase && _base == 16 ) {
148  *p_str = 'h';
149  *++p_str = '\0';
150  p_str--;
151  }
152  else {
153  *p_str = '\0';
154  }
155 
156  /* Move back along the number string (from right to left) and insert the
157  * digits on the way. In case of the number 123 the 3 will be inserted
158  * first, since this is the rightmost digit.
159  */
160  do {
161  *--p_str = digit[_num % _base];
162  _num /= _base;
163  } while( _num );
164 
165  Uart1_puts(str);
166 
167  return;
168 }

170 void
171 Uart1_putBits( const uint64_t _v, uint8_t _nBits ) {
172  uint8_t i;
173 
174  // Comparing with 8 instead of 7, due to overflow of unsigned.
175  for( i = _nBits; i > 0; i-- ) {
176  /* Output plus 48, since character '0' is in the 48th place of the
177  * ASCII table.
178  */
179  Uart1_putc( ( (_v >>(i-1)) & 1) + 48 );
180  }
181 
182  return;
183 }

208 void
209 Uart1_gets( char *_str, uint8_t _num ) {
210  uint8_t i = 0;
211  while(i < _num) {
212  if(pRxBuffer->bufferOverflowFlag == 1) {
213  /* Handle overflow error by (at this time) informing about the
214  * overflow.
215  */
216 #ifdef DEBUG
217  Uart1_printf("\n\n ERROR: %s:%d:%s(): " \
218  "UART1 FIFO software buffer overflow\n\n",
219  __FILE__, __LINE__, __FUNCTION__);
220 #else /* DEBUG */
221  Uart1_puts("\n\n ERROR: "); Uart1_puts(__FILE__);
222  Uart1_putNum(__LINE__, 10, false); Uart1_puts(__FUNCTION__);
223  Uart1_puts("UART1 FIFO software buffer overflow\n\n");
224 #endif /* DEBUG */
225 
226  return;
227  } else if(pRxBuffer->bufferNotEmptyFlag) {
228  *_str = swFifoBufferGet( pRxBuffer );
229 
230  if(*_str == UART_CR || *_str == UART_LF) {
231  *_str = '\0'; // Ignore character and end the string
232  return; // String is complete, exit loop and return
233  }
234 
235  _str++; // Move to next position in the string
236  }
237  i++;
238  }
239 
240  return;
241 }
