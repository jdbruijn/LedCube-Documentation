52 {

54 
55  // Check if there is room in the buffer
56  if (_pBuffer->numBytes == FIFO_BUFFER_SIZE) {
57  _pBuffer->bufferOverflowFlag = 1;
58  } else if (_pBuffer->numBytes < FIFO_BUFFER_SIZE) {
59  _pBuffer->data[_pBuffer->iLast] = _inData;
60 
61  _pBuffer->iLast++;
62  _pBuffer->numBytes++;
63  }
64

67  _pBuffer->bufferFullFlag = 1;
68  }
69

72  _pBuffer->iLast = 0;
73  }
