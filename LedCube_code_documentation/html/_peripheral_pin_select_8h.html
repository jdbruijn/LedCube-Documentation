#define PPS_UNLOCK
Value:
asm volatile ( "push w1 \n" \
"push w2 \n" \
"push w3 \n" \
"mov #OSCCON, w1 \n" \
"mov #0x46, w2 \n" \
"mov #0x57, w3 \n" \
"mov.b w2, [w1] \n" \
"mov.b w3, [w1] \n" \
"bclr OSCCON, #6 \n" \
"pop w3 \n" \
"pop w2 \n" \
"pop w1" )

Follow the necessary process to clear the IOLOCK bit in order to allow I/O mapping to be modified.

IOL1WAY in the Configuration Bits must be OFF

Note
Use inline assembly to unlock the Peripheral Pin Selection (PPS). First writing the unlock sequence (46h followed by 57h) to OSCCON<7:0> and than clear the IOLOCK bit in the OSCCON to disable the I/O Lock.

#define PPS_LOCK
Value:
asm volatile( "push w1 \n" \
"push w2 \n" \
"push w3 \n" \
"mov #OSCCON, w1 \n" \
"mov #0x46, w2 \n" \
"mov #0x57, w3 \n" \
"mov.b w2, [w1] \n" \
"mov.b w3, [w1] \n" \
"bset OSCCON, #6 \n" \
"pop w3 \n" \
"pop w2 \n" \
"pop w1" )

Follow the necessary process to set the IOLOCK bit to lock I/O mapping from being modified.

IOL1WAY in the Configuration Bits must be OFF

Note
Use inline assembly to lock the Peripheral Pin Selection (PPS). First writing the unlock sequence (46h followed by 57h) to OSCCON<7:0> and than set the IOLOCK bit in the OSCCON to enable the I/O Lock.
