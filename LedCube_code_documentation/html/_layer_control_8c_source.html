60 LayerControl_init(void)

62  /********** Setup PanelControl ********************************************/
63  PanelControl_init();

65  /********** Configure SPIx module *****************************************/
66  SPI2STAT = 0; // Reset SPIx module
67  SPI2CON1 = 0;
68  SPI2CON2 = 0;
69  Nop();
70 
71  IFS2bits.SPI2IF = 0; // Clear the Interrupt flag
72  IEC2bits.SPI2IE = 0; // Disable the interrupt
73 
74  SPI2CON1bits.MSTEN = 1; // Master mode
75  /* PPRE<1:0>: Primary Prescale bits (Master mode)
76  * 11 = Primary prescale 1:1
77  * 10 = Primary prescale 4:1
78  * 01 = Primary prescale 16:1
79  * 00 = Primary prescale 64:1
80  * @Note Do not set the primary and secondary prescalers to the
81  * value of 1:1 at the same time.
82  */
83  SPI2CON1bits.PPRE = 0b11;
84  /* SPRE<2:0>: Secondary Prescale bits (Master mode)
85  * 111 = Secondary prescale 1:1
86  * 110 = Secondary prescale 2:1
87  * 101 = Secondary prescale 3:1
88  * 100 = Secondary prescale 4:1
89  * 011 = Secondary prescale 5:1
90  * 010 = Secondary prescale 6:1
91  * 001 = Secondary prescale 7:1
92  * 000 = Secondary prescale 8:1
93  * @Note Do not set the primary and secondary prescalers to the
94  * value of 1:1 at the same time.
95  */
96  SPI2CON1bits.SPRE = 0b110;
97  /* CKE: SPIx Clock Edge Select bit
98  * 1 = Serial output data changes on transition from active clock state to
99  * Idle clock state (see CKP)
100  * 0 = Serial output data changes on transition from Idle clock state to
101  * active clock state (see CKP)
102  */
103  SPI2CON1bits.CKE = 1;
104  /* CKP: Clock Polarity Select bit
105  * 1 = Idle state for clock is a high level; active state is a low level
106  * 0 = Idle state for clock is a low level; active state is a high level
107  */
108  SPI2CON1bits.CKP = 0;
109 
110  {
111  uint8_t rData;
112  rData = SPI2BUF; // Clear the SPIx transmit/receive buffer
113  }
114 
115  SPI2STATbits.SPIEN = 1; // Enable SPIx module
116  Nop();
117 
118  /********** Configure Timer 3 *********************************************/
119  T3CON = 0; // Reset Timer x
120  TMR3 = 0; // Clear contents of Timer x register
121  Nop();
122 
123  /* @todo remove this block of comment
124  * Fpb of 16 MHz, TCKPS 8
125  * need period of 8000 to get 250 Hz
126  * 2000 to get 1 kHz
127  * 1600 to get 1.25 kHz
128  * 1000 to get 2 kHz
129  * 800 to get 2.5 kHz
130  *
131  * ////
132  * TCKPS 256
133  * need period of 62500 to get 1 Hz
134  * 31250 to get 2 Hz
135  * 6250 to get 10 Hz
136  * 3125 to get 20 Hz
137  * 2976 to get 21 Hz
138  * 2840 to get 22 Hz
139  * 2717 to get 23 Hz
140  * 2604 to get 24 Hz
141  */
142  PR3 = 1600; // Interrupt at 1 Hz @todo Final: set final frequency and edit comment

144  IFS0bits.T3IF = 0; // Clear the Timer x interrupt flag
145  IEC0bits.T3IE = 1; // Enable Timer x interrupts
146 
147  /* TCKPS<1:0>: Timerx Input Clock Prescale bits
148  * 11 = 1:256 prescaler value
149  * 10 = 1:64 prescaler value

166 void
167 LayerControl_setLayer(const uint8_t _layer)
168 {
169  DEBUG_PRINTF_FUNCTION_CALL("%u", _layer);
170 
171  SPI2_WAIT_TILL_TX_BUFFER_IS_EMPTY;
172 
173  /* Shift 0xFF7F, which is 1111 1111 0111 1111 binary, _layer places to the
174  * right so at all times one bit (layer) is 0 (turned on). Only a 8-bit
175  * value is needed but since it is shifted to the right an additional 8-bit
176  * value of 0xFF is needed for padding the value with ones. Since a logical
177  * 1 as output of the shift register results in 0 Volt on the MOSFET output.
178  * Finally AND it with 0xFF so it becomes an 8-bit value that can be put in
179  * the SPIx buffer.
180  */
181  //SPI2BUF = (0xFF7F >> _layer) & 0xFF;
182  SPI2BUF = layerReg[_layer];
183 
184  SPI2_WAIT_TILL_TX_BUFFER_IS_EMPTY;
185 
186  return;
187 }

247 _T3Interrupt(void)
248 {
249  LayerControl_setLayer(layer);
250 
251  LayerControl_update(pCubeControlData, pCubeControlData->pCubeDataRead,
252  layer, bamRound);
253 
254  // layer = (layer == LC_MAX_Z_C) ? 0 : layer + 1;

256  if (layer == LC_MAX_Z_C) {
257  layer = 0;
258  bamRound = (bamRound == LC_MAX_BAM_VAL) ? 0 : bamRound + 1;
259  } else {
260  layer++;
261  }
262 
263  PORT_SET_PIN(OE); // Disable the output during the update
264  PORT_PULSE_PIN(LD_LE);
265  LAYERCONTROL_PULSE_LATCH;
266  PORT_CLEAR_PIN(OE); // Enable the output again
267 
268  IFS0bits.T3IF = 0; // Clear the Timer x interrupt flag
269 }
