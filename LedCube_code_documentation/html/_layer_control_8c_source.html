153 LayerControl_setLayer( const uint8_t _layer ) {
154  DEBUG_PRINTF_FUNCTION_CALL("%u", _layer);
155 
156  SPI2_WaitTillTxBufferEmpty();
157 
158  /* Shift 0xFF7F, which is 1111 1111 0111 1111 binary, _layer places to the
159  * right so at all times one bit (layer) is 0 (turned on). Only a 8-bit
160  * value is needed but since it is shifted to the right an additional 8-bit
161  * value of 0xFF is needed for padding the value with ones. Since a logical
162  * 1 as output of the shift register results in 0 Volt on the MOSFET output.
163  * Finally AND it with 0xFF so it becomes an 8-bit value that can be put in
164  * the SPIx buffer.
165  */
166  //SPI2BUF = (0xFF7F >> _layer) & 0xFF;
167  SPI2BUF = layerReg[_layer];
168 
169  SPI2_WaitTillTxBufferEmpty();
170 
171  return;
172 }
