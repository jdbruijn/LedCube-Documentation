39 void
40 Buzzer_init(void)
41 {
42  /********** Configure Output Compare Module *******************************/
43  OC1CON = 0x0000; // Turn off Output Compare 1 Module
44 
45  PR2 = 499; // PWM period (4 kHz)
46  OC1RS = 0; // Initialize Secondary Compare register with 0% duty cycle
47  OC1R = 0; // Initialize Compare register with 0% duty cycle
48 
49  OC1CONbits.OCTSEL = 0; // Use timer 2 as clock source
50  OC1CONbits.OCM = 0b110; // PWM mode, Fault pin is disabled
51 
52  /********** Configure Timer 2 *********************************************/
53  T2CONbits.TON = 0; // Stop the timer during config
54  /* TCKPS<1:0>: Timerx Input Clock Prescale bits
55  * 11 = 1:256 prescaler value
56  * 10 = 1:64 prescaler value
57  * 01 = 1:8 prescaler value
58  * 00 = 1:1 prescaler value
59  */
60  T2CONbits.TCKPS = 0b01;
61  T2CONbits.TCS = 0; // Use Internal clock (Fosc/2) as clock source
62  T2CONbits.T32 = 0; // Form 2 seperate 16-bit timers
63  T2CONbits.TON = 1; // Enable timer
64 
65  DEBUG_PRINTF_FUNCTION_INITIALIZE_COMPLETE;
66  return;
67 }

95 void
96 Buzzer_set(uint16_t _frequency, uint32_t _time)
97 {
98  DEBUG_PRINTF_FUNCTION_CALL("%u, %lu", _frequency, _time);
99 
100  /********** Check conditions **********************************************/
101  ASSERT(_frequency >= BUZZER_MIN_FREQUENCY && \
102  _frequency <= BUZZER_MAX_FREQUENCY);
103  ASSERT(_time >= BUZZER_MIN_TIME && \
104  _time <= BUZZER_MAX_TIME);
105 
106  /********** Set frequency *************************************************/
107  /* "PWM Period = [(PRy + 1) * TCY * (TMRy Prescale Value)]
108  * PWM Frequency = 1/[PWM Period]" (Output Compare, Equation 4-1:
109  * Calculating the PWM Period, Microchip)
110  *
111  * PWM Period = 1/[PWM Frequency]
112  * PR = ((1 / _frequency) / (( 1/PBCLK) * Timer Prescaler)) -1
113  * = (1 / (_frequency * (Timer Prescaler / PBCLK))) - 1
114  */
115  PR2 = (1 / (_frequency * ((float) T2_TCKPS / FPB))) - 1;
116  OC1RS = (PR2 >> 1); // 50% duty cycle
117  DELAY_MS(_time);
118  OC1RS = 0; // 0% duty cycle (buzzer is turned off)
119 
120  return;
121 }
