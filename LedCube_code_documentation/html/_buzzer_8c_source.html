28 void
29 Buzzer_init( void ) {
30  /********** Configure Output Compare Module *******************************/
31  OC1CON = 0x0000; // Turn off Output Compare 1 Module
32 
33  PR2 = 499; // PWM period (4 kHz)
34  OC1RS = 0; // Initialize Secondary Compare register with 0% duty cycle
35  OC1R = 0; // Initialize Compare register with 0% duty cycle
36 
37  OC1CONbits.OCTSEL = 0; // Use timer 2 as clock source
38  OC1CONbits.OCM = 0b110; // PWM mode, Fault pin is disabled
39 
40  /********** Configure Timer 2 *********************************************/
41  T2CONbits.TON = 0; // Stop the timer during config
49  T2CONbits.TCKPS = 0b01;
50  T2CONbits.TCS = 0; // Use Internal clock (Fosc/2) as clock source
51  T2CONbits.T32 = 0; // Form 2 seperate 16-bit timers
52  T2CONbits.TON = 1; // Enable timer
53 
54  DEBUG_PRINTF_FUNCTION_INITIALIZE_COMPLETE();
55  return;
56 }

102  _time <= BUZZER_MAX_TIME );
103 
104  /********** Set frequency *************************************************/
114  PR2 = (1 / (_frequency * ((float)T2_TCKPS / FPB))) - 1;
115  OC1RS = (PR2 >> 1); // 50% duty cycle
