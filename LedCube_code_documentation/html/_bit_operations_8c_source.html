108 {
109  // swap odd and even bits
110  *_v = ((*_v >> 1) & 0x55555555) | ((*_v & 0x55555555) << 1);
111  // swap consecutive pairs
112  *_v = ((*_v >> 2) & 0x33333333) | ((*_v & 0x33333333) << 2);
113  // swap nibbles ...
114  *_v = ((*_v >> 4) & 0x0F0F0F0F) | ((*_v & 0x0F0F0F0F) << 4);
115  // swap bytes
116  *_v = ((*_v >> 8) & 0x00FF00FF) | ((*_v & 0x00FF00FF) << 8);
117  // swap 2-byte long pairs
118  *_v = (*_v >> 16) | (*_v << 16);
119
